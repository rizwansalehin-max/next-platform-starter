<script>
const letterMap = {
    'A':3, 'B':5, 'C':8, 'D':2, 'E':11, 'F':6, 'G':14, 'H':27,
    'I':31, 'J':22, 'K':17, 'L':19, 'M':21, 'N':18, 'O':29,
    'P':7, 'Q':15, 'R':24, 'S':9, 'T':4, 'U':18, 'V':25,
    'W':13, 'X':27, 'Y':16, 'Z':24
};

const punctSwap = {
    '!': ',', 
    ',': '!'
};

function convertEvans() {
    let text = document.getElementById('inputText').value.toUpperCase();
    let result = '';
    let prevCharLetter = false;

    for (let i = 0; i < text.length; i++) {
        let char = text[i];
        if (char >= 'A' && char <= 'Z') {
            if (prevCharLetter) {
                result += '*';
            }
            result += letterMap[char];
            prevCharLetter = true;
        } else {
            prevCharLetter = false;
            if (char === ' ') {
                continue; // No extra 0 for spaces (compact)
            } else if (char in punctSwap) {
                result += punctSwap[char];
            } else if (char === '.') {
                result += '?';
            } else if (char === '?') {
                result += '?';
            } else {
                result += '0'; // unsupported symbols
            }
        }
    }

    document.getElementById('outputText').innerText = result;
}
</script>
